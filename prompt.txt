from sklearn.metrics import (
    classification_report, confusion_matrix,
    roc_auc_score
)
from sklearn.preprocessing import label_binarize
import numpy as np

CLASS_NAMES = ['NONE', 'STRATEGIC', 'RELATIONAL', 'LEGAL', 'FINANCIAL']

def evaluate_multiclass(model, X_test, y_test, model_name):
    # Get predictions
    if hasattr(model, 'predict_proba'):
        y_prob = model.predict_proba(X_test)        # shape (N, 5)
        y_pred = np.argmax(y_prob, axis=1)
    else:
        y_pred = model.predict(X_test)
        y_prob = None

    print(f"\n=== {model_name} MULTICLASS RESULTS ===")
    print(classification_report(y_test, y_pred, target_names=CLASS_NAMES, zero_division=0))
    print("Confusion Matrix:")
    print(confusion_matrix(y_test, y_pred))

    # Per-class ROC-AUC (One-vs-Rest)
    if y_prob is not None:
        y_bin = label_binarize(y_test, classes=list(range(5)))
        for i, cls in enumerate(CLASS_NAMES):
            try:
                auc = roc_auc_score(y_bin[:, i], y_prob[:, i])
                print(f"ROC-AUC ({cls}): {auc:.4f}")
            except Exception:
                print(f"ROC-AUC ({cls}): N/A")

        macro_auc = roc_auc_score(y_bin, y_prob, multi_class='ovr', average='macro')
        print(f"Macro ROC-AUC: {macro_auc:.4f}")
